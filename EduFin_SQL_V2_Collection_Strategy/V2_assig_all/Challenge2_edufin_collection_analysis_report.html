C2-A: Employment Segmentation

Purpose: Group defaulters by employment type to understand the scale of each segment

-- C2-A: Basic employment segmentation with defaulter counts SELECT c.employment_type, COUNT(DISTINCT c.customer_id) AS total_defaulters, SUM(CASE WHEN l.loan_status = 'Defaulted' THEN l.loan_amount ELSE 0 END) AS total_collection_target, AVG(c.annual_income) AS avg_income, AVG(c.cibil_score) AS avg_cibil FROM customers c INNER JOIN loans l ON c.customer_id = l.customer_id WHERE l.loan_status = 'Defaulted' GROUP BY c.employment_type;
